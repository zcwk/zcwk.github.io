## 利用Nimble 流程控制工具简化代码

setTimeout(function(){
	console.log('我是1秒后执行');
	setTimeout(function(){
		console.log('我是0.5秒后执行');
		setTimeout(function(){
			console.log('我是0.1秒后执行');
		},100);
	},500);
},1000);

利用Nimble简化后的代码

先安装Nimble
npm install nimble

var flow = require('nimble');
flow.series([//建一个数组让它一个个的执行
function(callback){
	setTimeout(function(){
		console.log('我是1秒后执行');
		callback();
	},1000);
},
function(callback){
	setTimeout(function(){
		console.log('我是0.5秒后执行');
		callback();
	},500);
},
function(callback){
	setTimeout(function(){
		console.log('我是0.1秒后执行');
		callback();
	},100);
}
]);
